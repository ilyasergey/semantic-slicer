# Core MATLAB program syntax summary #


MProgram    ::=   MSeq

MSeq        ::=   MStmt {";" MStmt}

MStmt       ::=   Assign
                | IfStmt
                | Expr

Expr        ::=   AddExpr

AddExpr     ::=   MultExpr
                | AddExpr ("+" | "-") MultExpr

MultExpr    :: =  PowerExpr
                | MultExpr ("*" | "/") SimpleExpr

PowerExpr   ::=   SimpleExpr
                | PowerExpr "^" SimpleExpr

SimpleExpr  ::=   Var
                | FloatLiteral
                | IntegerLiteral
                | StringLiteral
                | "(" Expr ")"
